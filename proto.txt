Theory DWhile
Needs
  Designs UWhile UBase Closure Exists Forall Equal Impl AOI And Or Not EQV
END







Law asg_def BEGIN
 axiom
 ,
eqv CS (SUB [ ('x$,'e$)] asg,design NS (true,and CS (ITER CS and CS eq [x$' 'e$],ITER CS and CS eq [O$' 'O$])))
END

Law var_list_fusion BEGIN
 axiom
 ,
eqv CS (and CS (ITER CS and CS eq [x$' 'x$],ITER CS and CS eq [O$' 'O$]),ITER CS and CS eq [O$' 'O$])
END

Law II_def BEGIN
 axiom
 ,
eqv CS (II,design NS (true,ITER CS and CS eq [O$' 'O$]))
END

Law bot_def BEGIN
 axiom
 ,
eqv CS (bot,design NS (false,true))
END

Law top_def BEGIN
 axiom
 ,
eqv CS (top,design NS (true,false))
END

Conjecture asg_simple BEGIN
  eqv CS (SUB [('x,'e) ] asg,design NS (true,and CS (eq CS (x','e),ITER CS and CS eq [O$' 'O$])))
END

Conjecture asg_unchanged BEGIN
  eqv CS (SUB [('x,'e) ] asg,SUB [('x,'e) ('y,'y) ] asg)
END

Conjecture asg_seq_same BEGIN
  eqv CS (sqcmp NS (SUB [('x,'e) ] asg,SUB [('x,'f) ] asg),SUB [('x,SUB [('x,'e) ] 'f) ] asg)
, VREL (DYNCOV 'e BY 'O$) (DYNCOV 'f BY 'O$) (DYNCOV 'x BY 'O$)
END

Conjecture asg_seq_cond BEGIN
  eqv CS (sqcmp NS (SUB [('x,'e) ] asg,cond CS (P,'b,Q)),cond CS (sqcmp NS (SUB [('x,'e) ] asg,P),SUB [('x,'e) ] 'b,sqcmp NS (SUB [('x,'e) ] asg,Q)))
, VREL (DYNCOV P BY 'O$ O$') (DYNCOV Q BY 'O$ O$') (DYNCOV 'b BY 'O$) (DYNCOV 'e BY 'O$) (DYNCOV 'x BY 'O$)
END

Conjecture sqcmp_runit BEGIN
  eqv CS (sqcmp NS (R,II),R)
, VREL (DYNCOV II BY 'O$ O$') (DYNCOV R BY 'O$ O$')
END

Conjecture sqcmp_lunit BEGIN
  eqv CS (sqcmp NS (II,R),R)
, VREL (DYNCOV II BY 'O$ O$') (DYNCOV R BY 'O$ O$')
END

Conjecture bot_true BEGIN
  eqv CS (bot,true)
END

Conjecture top_nok BEGIN
  eqv CS (top,not CS ('ok))
END
