Theory UWhile
Needs
  UBase Closure Exists Forall Equal Impl AOI And Or Not EQV
END


Known asg : FUN t TO FUN t TO B ENDF ENDF .
Known cond : FUN B TO FUN B TO FUN B TO B ENDF ENDF ENDF .
Known sqcmp : FUN B TO FUN B TO B ENDF ENDF .

Known O$' :: set 


Law cond_def BEGIN
 axiom
 ,
eqv CS (cond CS (P,'b,Q),or CS (and CS ('b,P),and CS (not CS ('b),Q)))
END

Law sqcmp_def BEGIN
 axiom
 ,
eqv CS (sqcmp NS (P,Q),QS exists O$_0
  @  and CS (SUB [ (O$',O$'0)] P,SUB [ ('O$,O$'0)] Q))
,
VREL (DYNCOV 'P BY 'O$ O$') (DYNCOV 'Q BY 'O$ O$')
  FREE O$'0
END

Conjecture cond_idem BEGIN
  eqv CS (cond CS (P,'b,P),P)
END

Conjecture cond_symm BEGIN
  eqv CS (cond CS (P,'b,Q),cond CS (Q,not CS ('b),P))
END

Conjecture cond_assoc BEGIN
  eqv CS (cond CS (cond CS (P,'b,Q),'c,R),cond CS (P,and CS ('b,'c),cond CS (Q,'c,R)))
END

Conjecture cond_distr BEGIN
  eqv CS (cond CS (P,'b,cond CS (Q,'c,R)),cond CS (cond CS (P,'b,Q),'c,cond CS (P,'b,R)))
END

Conjecture cond_runit BEGIN
  eqv CS (cond CS (P,true,Q),P)
END

Conjecture cond_lunit BEGIN
  eqv CS (cond CS (Q,false,P),P)
END

Conjecture cond_absorb BEGIN
  eqv CS (cond CS (P,'b,cond CS (Q,'b,R)),cond CS (P,'b,R))
END

Conjecture cond_merge BEGIN
  eqv CS (cond CS (P,'b,cond CS (P,'c,Q)),cond CS (P,or CS ('b,'c),Q))
END

Conjecture cond_mdistr BEGIN
  eqv CS (cond CS (star CS (P,Q),'b,star CS (R,S)),star CS (cond CS (P,'b,R),cond CS (Q,'b,S)))
END

Conjecture cond_alt_def BEGIN
  eqv CS (cond CS (P,'b,Q),and CS (imp CS ('b,P),imp CS (not CS ('b),Q)))
END

Conjecture cond_alt_def2 BEGIN
  eqv CS (cond CS (P,'b,Q),eqv CS (or CS (not CS ('b),P),or CS ('b,Q)))
END

Conjecture sqcmp_assoc BEGIN
  eqv CS (sqcmp NS (P,sqcmp NS (Q,R)),sqcmp NS (sqcmp NS (P,Q),R))
, VREL (DYNCOV P BY 'O$ O$') (DYNCOV Q BY 'O$ O$') (DYNCOV R BY 'O$ O$')
END

Conjecture sqcmp_cond_l_distr BEGIN
  eqv CS (sqcmp NS (cond CS (P,'b,Q),R),cond CS (sqcmp NS (P,R),'b,sqcmp NS (Q,R)))
, VREL (DYNCOV P BY 'O$ O$') (DYNCOV Q BY 'O$ O$') (DYNCOV R BY 'O$ O$') (DYNCOV 'b BY 'O$)
END

Conjecture cond_sqcap_distr BEGIN
  eqv CS (cond CS (P,'b,sqcap CS (Q,R)),sqcap CS (cond CS (P,'b,Q),cond CS (P,'b,R)))
END

Conjecture sqcmp_sqcap_ldistr BEGIN
  eqv CS (sqcmp NS (sqcap CS (P,Q),R),sqcap CS (sqcmp NS (P,R),sqcmp NS (Q,R)))
, VREL (DYNCOV P BY 'O$ O$') (DYNCOV Q BY 'O$ O$') (DYNCOV R BY 'O$ O$')
END

Conjecture sqcmp_sqcap_rdistr BEGIN
  eqv CS (sqcmp NS (P,sqcap CS (Q,R)),sqcap CS (sqcmp NS (P,Q),sqcmp NS (P,R)))
, VREL (DYNCOV P BY 'O$ O$') (DYNCOV Q BY 'O$ O$') (DYNCOV R BY 'O$ O$')
END

Conjecture sqcap_cond_distr BEGIN
  eqv CS (sqcap CS (P,cond CS (Q,'b,R)),cond CS (sqcap CS (P,Q),'b,sqcap CS (P,R)))
END
